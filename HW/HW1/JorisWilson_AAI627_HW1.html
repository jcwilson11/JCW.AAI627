2. Prep features¶

In [3]:
# Convert to 1/0, since XGBoost expects numerical values
# s shows an importance of 90, while b shows an importance of 84
df = pd.read_csv("training.csv")
df["y"] = (df["Label"] == "s").astype(int)

DROP = ["EventId", "Label", "y", "Weight"]
X = df.drop(columns=[c for c in DROP if c in df.columns])
X = X.replace(-999, np.nan)  # let xgboost handle missing values
y = df["y"].values
w = df["Weight"].values if "Weight" in df.columns else np.ones(len(df))

X_train, X_valid, y_train, y_valid, w_train, w_valid = train_test_split(
    X, y, w, test_size=0.20, random_state=42, stratify=y
)
'''
neg = (y_train == 0).sum()
pos = (y_train == 1).sum()
scale_pos_weight = neg / max(pos, 1)
'''
w_pos = w_train[y_train == 1].sum()
w_neg = w_train[y_train == 0].sum()
scale_pos_weight = w_neg / max(w_pos, 1e-12)

from sklearn.model_selection import StratifiedKFold, RandomizedSearchCV
from xgboost import XGBClassifier
from sklearn.metrics import roc_auc_score

# 1) Define param distributions
param_distributions = {
    "n_estimators": [200, 300, 400, 600, 800, 1000, 1200, 1400, 1600],
    "learning_rate": [0.01, 0.03, 0.05, 0.07, 0.1, 0.15, 0.2, 0.25, 0.3, 0.4],
    "max_depth": [3, 4, 5, 6, 7, 8, 9, 10],
    "min_child_weight": [1, 2, 4, 6, 8, 10, 12, 14, 16],
    "gamma": [0, 0.5, 1, 2, 3, 5],
    "subsample": [0.5, 0.6, 0.7, 0.8, 0.9, 1.0],
    "colsample_bytree": [0.6, 0.7, 0.8, 0.9, 1.0],
    "reg_lambda": [0.5, 1, 2, 5, 10],
    "reg_alpha": [0, 0.1, 0.5, 1, 2, 5],
}

# 2) Base estimator for search
base_for_search = XGBClassifier(
    objective="binary:logistic",
    eval_metric="auc",
    tree_method="hist",
    scale_pos_weight=scale_pos_weight,   # keep fixed based on TRAIN split
    random_state=42,
    n_jobs=-1,
)

cv = StratifiedKFold(n_splits=6, shuffle=True, random_state=42)

search = RandomizedSearchCV(
    estimator=base_for_search,
    param_distributions=param_distributions,
    n_iter=60,                # increase for a deeper search
    scoring="roc_auc",
    n_jobs=-1,
    cv=cv,
    verbose=1,
    refit=True,               # refits best on the full TRAIN split
    random_state=42,
)

search.fit(X_train, y_train, sample_weight=w_train)

print("Best CV AUC:", search.best_score_)
print("Best params:", search.best_params_)

# 3) Final refit with early stopping on your held-out validation set
best_params = search.best_params_
final_model = XGBClassifier(
    objective="binary:logistic",
    eval_metric="auc",
    tree_method="hist",
    scale_pos_weight=scale_pos_weight,
    random_state=42,
    n_jobs=-1,
    **best_params,
    early_stopping_rounds=200, #moved from fit
)

final_model.fit(
    X_train, y_train,
    sample_weight=w_train,
    eval_set=[(X_valid, y_valid)],
    sample_weight_eval_set=[w_valid],
    
    verbose=False,
)

p_valid = final_model.predict_proba(X_valid)[:, 1]
print("Validation AUC after tuning + early stopping:", roc_auc_score(y_valid, p_valid, sample_weight=w_valid))

from sklearn.model_selection import GridSearchCV, StratifiedKFold
from xgboost import XGBClassifier
from sklearn.metrics import roc_auc_score

anchor = {
    'subsample': 0.7, 'reg_lambda': 0.5, 'reg_alpha': 0.0, 'n_estimators': 1400,
    'min_child_weight': 14, 'max_depth': 8, 'learning_rate': 0.01,
    'gamma': 2, 'colsample_bytree': 0.6,
}

param_grid_small = {
    "max_depth":        [7, 8, 9],        # 3
    "min_child_weight": [12, 14, 16],     # 3
    "subsample":        [0.6, 0.7, 0.8],  # 3
    "colsample_bytree": [0.6, 0.7],       # 2
}

xgb = XGBClassifier(
    objective="binary:logistic", eval_metric="auc", tree_method="hist",
    scale_pos_weight=scale_pos_weight, random_state=42, n_jobs=-1,
    # locked params stay at your anchor
    learning_rate=anchor["learning_rate"],
    n_estimators=anchor["n_estimators"],
    gamma=anchor["gamma"],
    reg_lambda=anchor["reg_lambda"],
    reg_alpha=anchor["reg_alpha"],
)

cv = StratifiedKFold(n_splits=3,shuffle=True, random_state=42)

grid = GridSearchCV(
    xgb, param_grid_small, scoring="roc_auc",
    n_jobs=-1, cv=cv, verbose=1, refit=True,
)
grid.fit(X_train, y_train, sample_weight=w_train)
